class Solution {
public:
    int triangularSum(vector<int>& nums) {
        int n = nums.size()-1;

        for(int k = n; k > 0; k--){ 
            for(int i = 0; i < k; i++){
                nums[i] = nums[i] +  nums[i+1] % 10;
                
            }
        }
        return nums[0] % 10;
    }
};
