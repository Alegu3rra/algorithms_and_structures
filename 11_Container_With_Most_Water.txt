class Solution {
public:
    int maxArea(vector<int>& height) {
        int areaWater = 0;
        int left = 0;
        int right = height.size() -1;

        while(left < right) {
            int temp = right - left;
            areaWater = max(areaWater, min(height[left], height[right])* temp);
            if(height[left] > height[right]){
                right--;
            } else{
                left++;
            }
        }
        return areaWater;
    }
};
